35/* Based on UNI3.1 standard by ATM Forum */
36
37/* ATM traffic types based on VPI=0 and (the following VCI */
38#define VCI_PPC 0x05 /* Point-to-point signal msg */
39#define VCI_BCC 0x02 /* Broadcast signal msg */
40#define VCI_OAMF4SC 0x03 /* Segment OAM F4 flow cell */
41#define VCI_OAMF4EC 0x04 /* End-to-end OAM F4 flow cell */
42#define VCI_METAC 0x01 /* Meta signal msg */
43#define VCI_ILMIC 0x10 /* ILMI msg */
44
45/* Q.2931 signalling messages */
46#define CALL_PROCEED 0x02 /* call proceeding */
47#define CONNECT 0x07 /* connect */
48#define CONNECT_ACK 0x0f /* connect_ack */
49#define SETUP 0x05 /* setup */
50#define RELEASE 0x4d /* release */
51#define RELEASE_DONE 0x5a /* release_done */
52#define RESTART 0x46 /* restart */
53#define RESTART_ACK 0x4e /* restart ack */
54#define STATUS 0x7d /* status */
55#define STATUS_ENQ 0x75 /* status ack */
56#define ADD_PARTY 0x80 /* add party */
57#define ADD_PARTY_ACK 0x81 /* add party ack */
58#define ADD_PARTY_REJ 0x82 /* add party rej */
59#define DROP_PARTY 0x83 /* drop party */
60#define DROP_PARTY_ACK 0x84 /* drop party ack */
61
62/* Information Element Parameters in the signalling messages */
63#define CAUSE 0x08 /* cause */
64#define ENDPT_REF 0x54 /* endpoint reference */
65#define AAL_PARA 0x58 /* ATM adaptation layer parameters */
66#define TRAFF_DESCRIP 0x59 /* atm traffic descriptors */
67#define CONNECT_ID 0x5a /* connection identifier */
68#define QOS_PARA 0x5c /* quality of service parameters */
69#define B_HIGHER 0x5d /* broadband higher layer information */
70#define B_BEARER 0x5e /* broadband bearer capability */
71#define B_LOWER 0x5f /* broadband lower information */
72#define CALLING_PARTY 0x6c /* calling party number */
73#define CALLED_PARTY 0x70 /* called party nmber */
74
75#define Q2931 0x09
76
77/* Q.2931 signalling general messages format */
78#define PROTO_POS 0 /* offset of protocol discriminator */
79#define CALL_REF_POS 2 /* offset of call reference value */
80#define MSG_TYPE_POS 5 /* offset of message type */
81#define MSG_LEN_POS 7 /* offset of mesage length */
82#define IE_BEGIN_POS 9 /* offset of first information element */
83
84/* format of signalling messages */
85#define TYPE_POS 0
86#define LEN_POS 2
87#define FIELD_BEGIN_POS 4
