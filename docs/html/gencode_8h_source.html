206/*
207 * A bit vector to represent definition sets. We assume TOT_REGISTERS
208 * is smaller than 8*sizeof(atomset).
209 */
210typedef bpf_u_int32 atomset;
211#define ATOMMASK(n) (1 << (n))
212#define ATOMELEM(d, n) (d & ATOMMASK(n))
