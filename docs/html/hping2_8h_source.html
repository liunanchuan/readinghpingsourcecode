281/*
282 * TCP header.
283 * Per RFC 793, September, 1981.
284 */
285struct mytcphdr {
286 __u16 th_sport; /* source port */
287 __u16 th_dport; /* destination port */
288 __u32 th_seq; /* sequence number */
289 __u32 th_ack; /* acknowledgement number */
290#if defined (__LITTLE_ENDIAN_BITFIELD)
291 __u8 th_x2:4, /* (unused) */
292 th_off:4; /* data offset */
293#elif defined (__BIG_ENDIAN_BITFIELD)
294 __u8 th_off:4, /* data offset */
295 th_x2:4; /* (unused) */
296#else
297#error "Please, edit Makefile and add -D__(LITTLE|BIG)_ENDIAN_BITFIEND"
298#endif
299 __u8 th_flags;
300 __u16 th_win; /* window */
301 __u16 th_sum; /* checksum */
302 __u16 th_urp; /* urgent pointer */
303};
